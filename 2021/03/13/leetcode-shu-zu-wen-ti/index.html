      for j:=i+1;j<len(array);j++{
         temp1 := array[i]
         temp2 := array[j]
         var temp = 0
         if temp1>temp2{
            temp=temp2
         }else {
            temp=temp1
         }
         area := (j-i)*temp
         if area>max {
            max = area
         }
      }
   }
   return max
}
