class Solution {
    public List<List<String>> solveNQueens(int n) {
        List<Integer[]> res = new ArrayList<>();
        int[] ans = new int[n];
        for(int i=0;i<n;i++){
            ans[i]=-1;
        }
        // ä»ç¬¬0è¡Œå¼€å§‹æ‘†Q
        getQueensRes(0,n,ans,res);
        // æ‰“å°ç»“æœ
        List<List<String>> result = new ArrayList<>();
        for(Integer[] ins : res){
            List<String> list = new ArrayList<>();
            for(int i=0;i<ins.length;i++){
                StringBuilder sb = new StringBuilder();
                for(int j=0;j<ins.length;j++){
                    if(ins[i]==j){
                        sb.append('Q');
                    }else {
                        sb.append('.');
                    }
                }
                list.add(sb.toString());
            }
            result.add(list);
        }
        return result;
    }

    private void getQueensRes(int row,int n,int[] ans,List<Integer[]> res){
        for(int j=0;j<n;j++){

        // åˆ¤æ–­ä¸€ä¸‹row>0ï¼Œå› ä¸ºç¬¬0è¡Œæ²¡æœ‰ä¸Šä¸€è¡Œ
        if(row>0){

    // åˆ¤æ–­å½“å‰ä½ç½®æ‘†Qæ˜¯å¦å’Œå‰é¢æ‘†è¿‡çš„Qä½ç½®æœ‰å†²çª
    private boolean canPutQueen(int row,int col,int[] ans){
        // å’Œå‰é¢æ‘†è¿‡çš„Qæ¯”è¾ƒ
        for(int i=0;i<row;i++){

            }
        }

class Solution {
    // 8çš‡å å †æ ˆæ³•
    public List<List<String>> solveNQueens3(int n) {
        List<Integer[]> res = new ArrayList<>();
        Stack<Queen> stack = new Stack<>();
        // ç”¨æ•°ç»„æ¥ç»´æŠ¤çš‡åçš„ä½ç½®
        int[] ans = new int[n];
        // åˆå§‹åŒ–ä¸º-1
        for(int i=0;i<n;i++){
            ans[i]=-1;
        }
        // ä»ç¬¬0è¡Œç¬¬0åˆ—å¼€å§‹æ‘†
        stack.add(new Queen(0,0));

        while (!stack.isEmpty()){
            Queen queen = stack.peek();
            // å¦‚æœå½“å‰ä½ç½®å¯ä»¥æ‘†Q
            if(canPutQueen3(queen.row,queen.col,ans)){
                ans[queen.row]=queen.col;
                // åˆ¤æ–­æ˜¯å¦åˆ°è¾¾æœ€åä¸€è¡Œ
                if(queen.row==n-1){
                    // è®°å½•ç»“æœ
                    Integer[] ins = new Integer[n];
                    for(int i=0;i<n;i++){
                        ins[i]=ans[i];
                    }
                    res.add(ins);
                    // å¼¹å‡ºé¡¶å±‚çš‡å,å›æº¯åˆ°ä¸Šä¸€è¡Œ
                    stack.pop();
                    // å¯èƒ½å½“å‰è¡Œæ˜¯ç¬¬0è¡Œ,stackä¼šä¸ºç©º
                    if(stack.isEmpty()){
                        break;
                    }
                    // æŠŠä¸Šä¸€è¡Œçš„Qç§»åŠ¨åˆ°ä¸‹ä¸€ä¸ªcolç»§ç»­æ‘†
                    Queen temp = stack.peek();
                    temp.col=temp.col+1;
                    // é‡ç½®ä½ç½®
                    ans[temp.row]=-1;
                    continue;
                }
                // ç§»åŠ¨æ‰¾ä¸‹ä¸€è¡Œç»§ç»­æ‘†
                stack.add(new Queen(queen.row+1,0));
            }else {
                // å¦‚æœcol=nè¯´æ˜è¿™è¡Œçš„æ‰€æœ‰colå·²ç»éå†å®Œä¸”éƒ½ä¸èƒ½æ‘†,è¯´æ˜ä¸Šä¸€è¡Œçš„Qæ‘†çš„ä¸å¯¹,éœ€è¦å›æº¯åˆ°ä¸Šä¸€è¡Œ,å¹¶ä»ä¸‹ä¸€ä¸ªcolå¼€å§‹ç»§ç»­å°è¯•
                if(queen.col==n){
                    // æŠŠå½“å‰è¡Œçš„Qå¼¹å‡º,å›æº¯åˆ°ä¸Šä¸€è¡Œ
                    stack.pop();
                    if(stack.isEmpty()){
                        break;
                    }
                    // æŠŠä¸Šä¸€è¡Œçš„Qç§»åŠ¨åˆ°ä¸‹ä¸€ä¸ªcolç»§ç»­æ‘†
                    Queen temp = stack.peek();
                    temp.col=temp.col+1;
                    // é‡ç½®ä½ç½®
                    ans[temp.row]=-1;
                    continue;
                }
                //å½“å‰è¡Œç»§ç»­æ‰¾ä¸‹ä¸€ä¸ªcol
                queen.col=queen.col+1;
            }
        }
        List<List<String>> result = new ArrayList<>();
        for(Integer[] integers : res){
            List<String> list = new ArrayList<>();
            for (Integer integer : integers) {
                StringBuilder sb = new StringBuilder();
                for (int m = 0; m < integers.length; m++) {
                    if (integer == m) {
                        sb.append('Q');
                    } else {
                        sb.append('.');
                    }
                }
                list.add(sb.toString());
            }
            result.add(list);
        }
        return result;

    }

    private boolean canPutQueen3(int row,int col,int[] ans){
        if(row==ans.length || col==ans.length){
            return false;
        }
        // å’Œå‰é¢æ‘†è¿‡çš„Qæ¯”è¾ƒ
        for(int i=0;i<row;i++){
            // åŒåˆ—
            if(ans[i]==col){
                return false;
            }
            // åŒæ–œçº¿
            if(Math.abs(ans[i]-col)==Math.abs(i-row)){
                return false;
            }
        }
        return true;
    }
}

class Queen {
    int row;
    int col;

    public Queen(int row,int col){
        this.row=row;
        this.col=col;
    }
}
